<script setup lang="ts">
import { computed } from 'vue'
import domtoimage from 'dom-to-image'
import TButton from './ui/TButton.vue'
import { useStore } from '~/composables/store'

const store = useStore()

const minWidth = computed({
  get: () => store.minWidth,
  set: value => store.minWidth = value,
})

const minHeight = computed({
  get: () => store.minHeight,
  set: value => store.minHeight = value,
})

const captureTreeMap = () => {
  const projectMapElement = document.querySelector('.treemap') as HTMLElement | null

  if (projectMapElement) {
    window.scrollTo(0, 0)
    nextTick(() => {
      domtoimage.toPng(projectMapElement, { quality: 1, bgcolor: 'white' })
        .then((dataUrl: string) => {
          const link = document.createElement('a')
          link.href = dataUrl
          link.download = `treemap-pulse-${Date.now()}.png`
          link.click()
        })
        .catch((error: Error) => {
          console.error('Error: ', error)
        })
    })
  }
}
</script>
